scripted_trigger losp_01_trigger = {
		is_alive = yes
		is_imprisoned = no
		is_female = yes
		is_totally_healthy = yes
		age >= 20
		is_pregnant = no
		is_commanding_army = no
}

losp.10 = {
	type = character_event
	title = losp.10.t
	theme = party	
	desc = losp.10.desc

	left_portrait = {
		character = scope:actor
		animation = happiness
	}
	right_portrait = {
		character = scope:losp_sexpartner_1
		animation = happiness
	}
	
	immediate = {
		play_music_cue = "mx_cue_seduction"
		
		scope:recipient = {
			random_consort = {
				limit = {
					losp_01_trigger = yes
					age <= 40
					NOT = { is_child_of = scope:actor }
				}
				save_scope_as = losp_sexpartner_1	
			}
			remove_character_flag = has_approximate_spouse		
		}
	}
	
	option = { #your lover never refuse you
		trigger = {
			scope:losp_sexpartner_1 = {
				has_relation_lover = scope:actor
			}
		}
		name = losp.10.c
		custom_tooltip = LOS_NEVER_REFUSE
		trigger_event = losp.20
	}
	
	option = { #sedece
		trigger = {
			NOT = {
				scope:losp_sexpartner_1 = {
					has_relation_lover = scope:actor
				}
			}
		}		
		name = losp.10.a
		random_list = {			
			80 = { 
				desc = LOSP_ACCEPT_SEDUCTION
				show_chance = no
				custom_tooltip = LOSP_ACCEPT_SEDUCTION.a

			}

				desc = LOSP_REFUSE_SEDUCTION
