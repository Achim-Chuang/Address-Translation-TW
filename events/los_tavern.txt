namespace = lost

scripted_trigger lost_02_trigger = {
	age <= 35
	age >= 20
	is_pregnant = no
	is_totally_healthy = yes
	is_female = yes
	is_married = yes
	is_alive = yes
	is_imprisoned = no
	is_commanding_army = no
	NOT = { is_child_of = scope:actor }
	NOT = { is_consort_of = scope:actor }
}

#visit tavern
lost.01 = {
	type = character_event
	title = lost.01.t
	desc = lost.01.desc
	theme = party
	
	override_background = {
        event_background = tavern
    }
	
	immediate = { 
		root = { save_scope_as = actor } 
		every_vassal_or_below = {
			if = {
				limit = {
					lost_02_trigger = yes
				}				
				add_to_temporary_list = potential_female_list
			}
			else_if = {
				limit ={
					is_female = no
					is_married = yes
					is_alive = yes
				}
				every_consort = {
					limit = {
						lost_02_trigger = yes
					}
					add_to_temporary_list = potential_female_list
				}
			}
		}
		every_neighboring_top_liege_realm_owner = {
			every_vassal_or_below = {
				if = {
					limit = {
						lost_02_trigger = yes
					}				
					add_to_temporary_list = potential_female_list
				}
				else_if = {
					limit = {
						is_female = no
						is_alive = yes
						is_married = yes
					}
					every_consort = {
						limit = {
							lost_02_trigger = yes
						}
						add_to_temporary_list = potential_female_list
					}
				}
			}
		}		
		random_in_list = {
			list = potential_female_list
			weight = {
				base = 100
				compare_modifier = {
					value = ai_sociability
				}				
				modifier = {
					add = 100
					has_trait = lustful
				}
			}
			save_scope_as = lost_potential_partner
		}		
	}
	option = {
		trigger = {
			NOT = { has_character_flag = los_ask_for_someone }
		}
		name = lost.01.a
		custom_tooltip = lost.01.a.t
		trigger_event = lost.10
	}
	option = {
		name = lost.01.b
		custom_tooltip = lost.01.b.t
		if = {
			limit = { exists = scope:lost_potential_partner }
			trigger_event = lost.40
		}
		else = { trigger_event = lost.04 }
	}
}
